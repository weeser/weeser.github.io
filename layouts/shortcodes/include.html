{{/* From https://discourse.gohugo.io/t/include-file-shortcode-for-including-files-with-shortcode-calls/16699 */}} {{ $file := .Get "file" }} {{ $type := .Get "type" }} {{ $language := .Get "language" }} {{ $options :=.Get "options" }}
{{- if eq $type "page" -}} {{- with .Site.GetPage $file }} {{- .Content -}} {{- else -}} {{- (errorf `%s: include shortcode could not find page named "%s"` .Position $file) -}} {{- end -}} {{- else -}} {{- if (.Get "language") -}} {{- highlight ($file | readFile) $language (default "linenos=table" $options) -}} {{- else if eq $type "html" -}} {{- $file | readFile | safeHTML -}} {{- else -}} {{- $file | readFile | $.Page.RenderString -}} {{- end -}} {{- end -}}
